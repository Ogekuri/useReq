 196def parse_version_tuple(version: str) -> tuple[int, ...] | None:
 197    """Converts a version into a numeric tuple for comparison.
 198
 199    Accepts versions in 'X.Y.Z' format (ignoring any non-numeric suffixes).
 200    """
 201
 202    cleaned = (version or "").strip()
 203    if not cleaned:
 204        return None
 205
 206    parts = cleaned.split(".")
 207    numbers: list[int] = []
 208    for part in parts:
 209        match = re.match(r"^(\d+)", part)
 210        if not match:
 211            break
 212        try:
 213            numbers.append(int(match.group(1)))
 214        except ValueError:
 215            return None
 216
 217    return tuple(numbers) if numbers else None

 220def is_newer_version(current: str, latest: str) -> bool:
 221    """Returns True if latest is greater than current."""
 222    current_tuple = parse_version_tuple(current)
 223    latest_tuple = parse_version_tuple(latest)
 224    if not current_tuple or not latest_tuple:
 225        return False
 226
 227    max_len = max(len(current_tuple), len(latest_tuple))
 228    current_norm = current_tuple + (0,) * (max_len - len(current_tuple))
 229    latest_norm = latest_tuple + (0,) * (max_len - len(latest_tuple))
 230    return latest_norm > current_norm

 233def maybe_notify_newer_version(timeout_seconds: float = 1.0) -> None:
 234    """Checks online for a new version and prints a warning.
 235
 236    If the call fails or the response is invalid, it prints nothing and proceeds.
 237    """
 238
 239    current_version = load_package_version()
 240    url = "https://api.github.com/repos/Ogekuri/useReq/releases/latest"
 241
 242    try:
 243        request = urllib.request.Request(
 244            url,
 245            headers={
 246                "Accept": "application/vnd.github+json",
 247                "User-Agent": f"usereq/{current_version}",
 248            },
 249            method="GET",
 250        )
 251        with urllib.request.urlopen(request, timeout=timeout_seconds) as response:
 252            body = response.read().decode("utf-8", errors="replace")
 253        payload = json.loads(body)
 254        tag = payload.get("tag_name")
 255        if not isinstance(tag, str) or not tag.strip():
 256            return
 257
 258        latest_version = normalize_release_tag(tag)
 259        if is_newer_version(current_version, latest_version):
 260            print(
 261                "A new version of usereq is available: "
 262                f"current {current_version}, latest {latest_version}. "
 263                "To upgrade, run: req --upgrade"
 264            )

 266        urllib.error.URLError,
 267        urllib.error.HTTPError,
 268        TimeoutError,
 269        json.JSONDecodeError,
 270        ValueError,

 292def make_relative_if_contains_project(path_value: str, project_base: Path) -> str:
 293    """Normalizes the path relative to the project root when possible."""
 294    if not path_value:
 295        return ""
 296    candidate = Path(path_value)
 297    if not candidate.is_absolute():
 298        parts = candidate.parts
 299        if parts and parts[0] == project_base.name and len(parts) > 1:
 300            candidate = Path(*parts[1:])
 301    if candidate.is_absolute():
 302        try:
 303            return str(candidate.relative_to(project_base))
 304        except ValueError:
 305            return str(candidate)
 306    resolved = (project_base / candidate).resolve(strict=False)
 307    try:
 308        return str(resolved.relative_to(project_base))
 309    except ValueError:
 310        pass
 311    base_str = str(project_base)
 312    if path_value.startswith(base_str):
 313        trimmed = path_value[len(base_str) :].lstrip("/\\")
 314        return trimmed
 315    return path_value

 468def normalize_description(value: str) -> str:
 469    """Normalizes a description by removing superfluous quotes and escapes."""
 470    trimmed = value.strip()
 471    if len(trimmed) >= 2 and trimmed.startswith('"') and trimmed.endswith('"'):
 472        trimmed = trimmed[1:-1]
 473    if len(trimmed) >= 4 and trimmed.startswith('\\"') and trimmed.endswith('\\"'):
 474        trimmed = trimmed[2:-2]
 475    return trimmed.replace('\\"', '"')

 478def md_to_toml(md_path: Path, toml_path: Path, force: bool) -> None:
 479    """Converts a Markdown prompt to TOML for Gemini."""
 480    if toml_path.exists() and not force:
 481        raise ReqError(
 482            f"Destination TOML already exists (use --force to overwrite): {toml_path}",
 483            3,
 484        )
 485    content = md_path.read_text(encoding="utf-8")
 486    match = re.match(r"^\s*---\s*\n(.*?)\n---\s*\n(.*)$", content, re.S)
 487    if not match:
 488        raise ReqError("No leading '---' block found at start of Markdown file.", 4)
 489    frontmatter, rest = match.groups()
 490    desc = extract_description(frontmatter)
 491    desc_escaped = desc.replace("\\", "\\\\").replace('"', '\\"')
 492    rest_text = rest if rest.endswith("\n") else rest + "\n"
 493    toml_body = [
 494        f'description = "{desc_escaped}"',
 495        "",
 496        'prompt = """',
 497        rest_text,
 498        '"""',
 499        "",
 500    ]
 501    toml_path.parent.mkdir(parents=True, exist_ok=True)
 502    toml_path.write_text("\n".join(toml_body), encoding="utf-8")
 503    dlog(f"Wrote TOML to: {toml_path}")

 531def extract_purpose_first_bullet(body: str) -> str:
 532    """Returns the first bullet of the Purpose section."""
 533    lines = body.splitlines()
 534    start_idx = None
 535    for idx, line in enumerate(lines):
 536        if line.strip().lower() == "## purpose":
 537            start_idx = idx + 1
 538            break
 539    if start_idx is None:
 540        raise ReqError("Error: missing '## Purpose' section in prompt.", 7)
 541    for line in lines[start_idx:]:
 542        stripped = line.strip()
 543        if stripped.startswith("#"):
 544            break
 545        match = re.match(r"^\s*-\s+(.*)$", line)
 546        if match:
 547            return match.group(1).strip()
 548    raise ReqError("Error: no bullet found under the '## Purpose' section.", 7)

 578def render_kiro_agent(
 579    template: str,
 580    name: str,
 581    description: str,
 582    prompt: str,
 583    resources: list[str],
 584    tools: list[str] | None = None,
 585    model: Optional[str] = None,
 586    include_tools: bool = False,
 587    include_model: bool = False,
 588) -> str:
 589    """Renders the Kiro agent JSON and populates main fields."""
 590    replacements = {
 591        "%%NAME%%": json_escape(name),
 592        "%%DESCRIPTION%%": json_escape(description),
 593        "%%PROMPT%%": json_escape(prompt),
 594    }
 595    if "%%RESOURCES%%" in template:
 596        resources_block = ",\n".join(f'    "{json_escape(item)}"' for item in resources)
 597        replacements["%%RESOURCES%%"] = resources_block
 598    for token, replacement in replacements.items():
 599        template = template.replace(token, replacement)
 600    try:
 601        parsed = json.loads(template)
 602        parsed["resources"] = resources
 603        if include_model and model is not None:
 604            parsed["model"] = model
 605        else:
 606            parsed.pop("model", None)
 607        if include_tools:
 608            parsed_tools = tools if isinstance(tools, list) else []
 609            parsed["tools"] = parsed_tools
 610            parsed["allowedTools"] = parsed_tools
 611        else:
 612            parsed.pop("tools", None)
 613            parsed.pop("allowedTools", None)
 614        return json.dumps(parsed, indent=2, ensure_ascii=False) + "\n"
 615    except Exception:
 616        # If parsing fails, return raw template to preserve previous behavior
 617        return template

 679def strip_json_comments(text: str) -> str:
 680    """Removes // and /* */ comments to allow JSONC parsing."""
 681    cleaned: list[str] = []
 682    in_block = False
 683    for line in text.splitlines():
 684        stripped = line.lstrip()
 685        if in_block:
 686            if "*/" in stripped:
 687                in_block = False
 688            continue
 689        if stripped.startswith("/*"):
 690            if "*/" not in stripped:
 691                in_block = True
 692            continue
 693        if stripped.startswith("//"):
 694            continue
 695        cleaned.append(line)
 696    return "\n".join(cleaned)

 729def get_model_tools_for_prompt(
 730    config: dict[str, Any] | None, prompt_name: str, source_name: Optional[str] = None
 731) -> tuple[Optional[str], Optional[list[str]]]:
 732    """Extracts model and tools for the prompt from the CLI config.
 733
 734    Returns (model, tools) where each value can be None if not available.
 735    """
 736    if not config:
 737        return None, None
 738    prompts = config.get("prompts") or {}
 739    entry = prompts.get(prompt_name) if isinstance(prompts, dict) else None
 740    model = None
 741    tools: Optional[list[str]] = None
 742    if isinstance(entry, dict):
 743        model = entry.get("model")
 744        mode = entry.get("mode")
 745        if mode:
 746            usage = config.get("usage_modes") or {}
 747            mode_entry = usage.get(mode) if isinstance(usage, dict) else None
 748            if isinstance(mode_entry, dict):
 749                # Use the unified key name 'tools' across all CLI configs.
 750                # Accept either a list of strings or a CSV string in the config.json.
 751                key_name = "tools"
 752                raw = mode_entry.get(key_name)
 753                if raw is None:
 754                    tools = None
 755                elif isinstance(raw, list):
 756                    tools = [str(t) for t in raw]
 757                elif isinstance(raw, str):
 758                    # Parse comma-separated string into list
 759                    items = [p.strip() for p in raw.split(",") if p.strip()]
 760                    tools = items if items else None
 761                else:
 762                    tools = None
 763    return model, tools

 766def get_raw_tools_for_prompt(config: dict[str, Any] | None, prompt_name: str) -> Any:
 767    """Returns the raw value of `usage_modes[mode]['tools']` for the prompt.
 768
 769    Can return a list of strings, a string, or None depending on how it is
 770    defined in `config.json`. Does not perform CSV parsing: returns the value
 771    exactly as present in the configuration file.
 772    """
 773    if not config:
 774        return None
 775    prompts = config.get("prompts") or {}
 776    entry = prompts.get(prompt_name) if isinstance(prompts, dict) else None
 777    if isinstance(entry, dict):
 778        mode = entry.get("mode")
 779        if mode:
 780            usage = config.get("usage_modes") or {}
 781            mode_entry = usage.get(mode) if isinstance(usage, dict) else None
 782            if isinstance(mode_entry, dict):
 783                return mode_entry.get("tools")
 784    return None

 794def deep_merge_dict(base: dict[str, Any], incoming: dict[str, Any]) -> dict[str, Any]:
 795    """Recursively merges dictionaries, prioritizing incoming values."""
 796    for key, value in incoming.items():
 797        if isinstance(value, dict) and isinstance(base.get(key), dict):
 798            deep_merge_dict(base[key], value)
 799        else:
 800            base[key] = value
 801    return base
